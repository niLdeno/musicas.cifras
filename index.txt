        function renderizarBracoInterativo() {
            const container = document.getElementById('braco-interativo-render');
            // atualiza configurações a partir dos inputs
            const startEl = document.getElementById('cad-start-fret');
            const numEl = document.getElementById('cad-num-casas');
            braco_start_fret = startEl ? parseInt(startEl.value || '1') : braco_start_fret;
            braco_num_casas = numEl ? parseInt(numEl.value || '5') : braco_num_casas;

            let html = '<div class="status-header">';

            // Renderiza cabeçalho (O/X)
            for (let i = 0; i < 6; i++) {
                let status = '';
                if (cordasAcordeAtual[i] === -1) {
                    status = '<span style="color: #ef4444;">X</span>';
                } else if (cordasAcordeAtual[i] === 0) {
                    status = '<span style="border: 2px solid var(--text-main); width: 14px; height: 14px; border-radius: 50%; display: inline-block;"></span>';
                }
                html += `<div class="status-item" onclick="toggleStatusCorda(${i})">${status}</div>`;
            }

            html += '</div><div class="pestana"></div><div class="braco-grid"><div class="cordas-linhas">';

            // Fundo: Linhas verticais das cordas
            for (let i = 0; i < 6; i++) html += '<div></div>';
            html += '</div>';

            // Células interativas das casas (start .. start+num-1)
            for (let idx = 0; idx < braco_num_casas; idx++) {
                const casaAbs = braco_start_fret + idx;
                for (let corda = 0; corda < 6; corda++) {
                    let temDedo = cordasAcordeAtual[corda] === casaAbs;
                    let dedoNum = dedosAcordeAtual[corda];
                    html += `<div class="celula-casa" onclick="marcarCasa(${corda}, ${idx})">
                                ${temDedo ? `<div class="dedo-press">${dedoNum ? dedoNum : ''}</div>` : ''}
                             </div>`;
                }
            }
            html += '</div>';
            container.innerHTML = html;
        }

        function toggleStatusCorda(corda) {
            // Se estava Muta (-1), fica Solta (0). Se era outra coisa, fica Muta (-1).
            if (cordasAcordeAtual[corda] === -1) cordasAcordeAtual[corda] = 0;
            else cordasAcordeAtual[corda] = -1;
            if (cordasAcordeAtual[corda] !== -1 && cordasAcordeAtual[corda] !== 0) {
                // keep finger
            } else {
                dedosAcordeAtual[corda] = null;
            }
            renderizarBracoInterativo();
        }

        function marcarCasa(corda, casaIdx) {
            // casaIdx é o índice visual (0..num-1). Converte para casa absoluta.
            const casaAbs = braco_start_fret + casaIdx;
            // Se clicar onde já tem dedo, a corda fica solta (0). Caso contrário, marca a casa absoluta.
            if (cordasAcordeAtual[corda] === casaAbs) {
                cordasAcordeAtual[corda] = 0;
                dedosAcordeAtual[corda] = null;
            } else {
                cordasAcordeAtual[corda] = casaAbs;
                // Pergunta pelo número do dedo (1-4) — vazio para nenhum
                let dedo = prompt('Número do dedo para essa posição (1-4). Deixe vazio para nenhum:', '');
                if (dedo !== null) {
                    dedo = dedo.toString().trim();
                    if (dedo === '') dedosAcordeAtual[corda] = null;
                    else {
                        const n = parseInt(dedo);
                        if (!isNaN(n) && n >= 1 && n <= 4) dedosAcordeAtual[corda] = n;
                        else dedosAcordeAtual[corda] = null;
                    }
                }
            }
            renderizarBracoInterativo();
        }

        async function salvarAcorde() {
            const nome = document.getElementById('cad-acorde-nome').value.trim();
            if (!nome) return alert("Dê um nome ao acorde.");

            // Usa o Array diretamente gerado pelos cliques no violão
            const posicoes = [...cordasAcordeAtual];
            const dedos = [...dedosAcordeAtual];

            if (!supabaseClient) {
                alert('Sem conexão ao Supabase. Abra "Conexão" e informe Project URL e API Key antes de salvar.');
                return;
            }

            // Garantir arrays com 6 elementos
            while (posicoes.length < 6) posicoes.push(0);
            while (dedos.length < 6) dedos.push(null);

            try {
                // Inclui metadados de exibição do braço
                const startFretToSave = parseInt(document.getElementById('cad-start-fret')?.value || braco_start_fret || 1);
                const numCasasToSave = parseInt(document.getElementById('cad-num-casas')?.value || braco_num_casas || 5);
                console.log('Salvando acorde', { nome, posicoes, dedos, start_fret: startFretToSave, num_casas: numCasasToSave });
                const { data, error } = await supabaseClient.from('dicionario_acordes').insert([{ nome, posicoes, dedos, start_fret: startFretToSave, num_casas: numCasasToSave }]).select();
                if (error) {
                    console.error('Erro ao salvar acorde:', error);
                    // Detecta erro de coluna 'dedos' ausente no schema cache do Supabase
                    const msg = (error.message || '').toString();
                    if (msg.includes("Could not find the 'dedos' column") || msg.includes("column \"dedos\" does not exist") || msg.includes('start_fret') || msg.includes('num_casas')) {
                        // Tenta gravar sem as colunas extras (compatibilidade)
                        const { data: data2, error: error2 } = await supabaseClient.from('dicionario_acordes').insert([{ nome, posicoes }]).select();
                        if (error2) {
                            console.error('Erro ao salvar sem dedos:', error2);
                            alert('Erro ao salvar acorde: ' + (error2.message || JSON.stringify(error2)) + '\nVerifique permissões do banco.');
                            return;
                        }
                        alert('Algumas colunas extras (por exemplo `dedos`, `start_fret` ou `num_casas`) não existem no banco; acorde salvo sem esses metadados.\nPara suportar completamente, adicione as colunas `dedos jsonb`, `start_fret int` e `num_casas int` na tabela `dicionario_acordes` via o editor SQL do Supabase.');
                        fecharModal('modal-cadastro-acorde');
                        carregarDicionario();
                        return;
                    }
                    alert('Erro ao salvar acorde: ' + (error.message || JSON.stringify(error)) + '\nVerifique a conexão e as permissões do banco.');
                    return;
                }
                // Sucesso
                fecharModal('modal-cadastro-acorde');
                carregarDicionario();
            } catch (err) {
                console.error('Exceção ao salvar acorde:', err);
                alert('Exceção ao salvar acorde. Veja o console para mais detalhes.');
            }
        }

        // === GERADOR DE DIAGRAMA SVG (VISOR) ===
        function desenharAcordeSVG(nomeAcorde, data, opts) {
            // `data` pode ser um array (posicoes) ou um objeto {posicoes, dedos}
            let posicoes = [], dedos = null;
            if (Array.isArray(data)) posicoes = data;
            else if (data && data.posicoes) { posicoes = data.posicoes; dedos = data.dedos || null; }
            const showFret = localStorage.getItem('gerecifras_show_fret_numbers') === '1';
            const globalStartFret = parseInt(localStorage.getItem('gerecifras_start_fret') || '1');
            const showFinger = localStorage.getItem('gerecifras_show_finger_numbers') === '1';

            const startFret = (data && data.start_fret) ? parseInt(data.start_fret) : globalStartFret;
            const numCasas = (data && data.num_casas) ? parseInt(data.num_casas) : (opts && opts.numCasas) || 5;
            const width = (opts && opts.width) || 140;
            const height = (opts && opts.height) || 200;
            const left = 20;
            const right = width - 20;
            const top = 40;
            const rowH = Math.floor((height - top - 20) / (numCasas + 1));

            let svg = `<svg width="${width}" height="${height}" viewBox="0 0 ${width} ${height}" xmlns="http://www.w3.org/2000/svg">`;
            svg += `<text x="${width / 2}" y="22" font-family="sans-serif" font-size="16" font-weight="bold" fill="var(--text-main)" text-anchor="middle">${nomeAcorde}</text>`;
            svg += `<line x1="${left}" y1="${top}" x2="${right}" y2="${top}" stroke="var(--text-main)" stroke-width="4"/>`;
            for (let i = 1; i <= numCasas; i++) { svg += `<line x1="${left}" y1="${top + i * rowH}" x2="${right}" y2="${top + i * rowH}" stroke="var(--text-muted)" stroke-width="1"/>`; }
            for (let i = 0; i < 6; i++) { let x = left + i * ((right - left) / 5); svg += `<line x1="${x}" y1="${top}" x2="${x}" y2="${top + numCasas * rowH}" stroke="var(--text-main)" stroke-width="1"/>`; }

            // Numeração das casas à esquerda
            if (showFret) {
                for (let i = 0; i < numCasas; i++) {
                    const numero = startFret + i;
                    const y = top + (i + 1) * rowH - Math.floor(rowH / 2);
                    svg += `<text x="${left - 8}" y="${y}" font-family="sans-serif" font-size="11" fill="var(--text-muted)" text-anchor="end">${numero}</text>`;
                }
            }

            for (let i = 0; i < 6; i++) {
                let x = left + i * ((right - left) / 5); let casa = (posicoes && posicoes[i] !== undefined) ? posicoes[i] : null;
                if (casa === -1) { svg += `<text x="${x}" y="${top - 6}" font-family="sans-serif" font-size="10" font-weight="bold" fill="#ef4444" text-anchor="middle">X</text>`; }
                else if (casa === 0) { svg += `<circle cx="${x}" cy="${top - 6}" r="3" fill="none" stroke="var(--text-main)"/>`; }
                else if (casa > 0) {
                    const rel = casa - startFret; // 0-based
                    const y = top + (rel + 1) * rowH - Math.floor(rowH / 2);
                    svg += `<circle cx="${x}" cy="${y}" r="${Math.max(5, Math.floor(rowH / 3))}" fill="var(--primary)"/>`;
                    if (showFinger && dedos && dedos[i]) { svg += `<text x="${x}" y="${y + 3}" font-family="sans-serif" font-size="${Math.max(10, Math.floor(rowH / 3))}" font-weight="700" fill="#fff" text-anchor="middle">${dedos[i]}</text>`; }
                }
            }
            svg += `</svg>`;
            return svg;
        }

        function mostrarAcorde(event, acorde) {
            event.stopPropagation();
            const tooltip = document.getElementById('tooltip-acorde');
            const data = obterPosicoesAcorde(acorde);

            if (data && data.posicoes) { tooltip.innerHTML = desenharAcordeSVG(acorde, data); }
            else { tooltip.innerHTML = `<div style="text-align:center; font-family:sans-serif; font-size:13px; color:var(--text-muted);">Diagrama<br>de <b>${acorde}</b><br>não mapeado.</div>`; }

            tooltip.style.left = (event.pageX - 50) + 'px'; tooltip.style.top = (event.pageY + 20) + 'px';
            tooltip.style.display = 'block';
            setTimeout(() => { document.addEventListener('click', fecharTooltip); }, 10);
        }

            });
